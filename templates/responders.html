<script>

function showUnHandled() {

  const rows = document.querySelectorAll(".responder-row");

  rows.forEach(r => {

    if (r.dataset.handled === "true") {

      r.style.display = "none";

    } else {

      r.style.display = "";

    }

  });

  document.getElementById("filterStatus").innerText = "（未対応のみ表示中）";

}


function showAll() {

  const rows = document.querySelectorAll(".responder-row");

  rows.forEach(r => r.style.display = "");

  document.getElementById("filterStatus").innerText = "（全件表示）";

}

</script>
